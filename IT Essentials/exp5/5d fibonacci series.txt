<script>
function fib()
{
var n=1,n1=0,n2=1,n3,st="";
var i=parseInt(document.getElementById("num").value);
if (i>0)
{st=st+n1;
n++;}
if(i>1)
{st=st+','+n2;
n++;}
while(n<=i)
{n3=n2+n1;
n1=n2;
n2=n3;
st=st+','+n3;
n++;}
document.getElementById("x").innerHTML=st;
}
